            function hexToRgb(hex) {

                var result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);

            }

            function getVolumetricColor(lum, style) {

                let palette;

                let stops; 

                if (style === 'volumetric_custom') {

                    palette = [ hexToRgb(customColorDark.value), hexToRgb(customColorMid.value), hexToRgb(customColorLight.value) ];

                    stops = [0.0, 0.45, 0.80];

                } else {

                    const palettes = {

                        'volumetric_blue': [[0,0,0], [0,20,80], [160,200,255], [255,255,255]], 

                        'volumetric_mono': [[0,0,0], [80,80,80], [180,180,180], [255,255,255]]

                    };

                    palette = palettes[style];

                    stops = [0.0, 0.25, 0.6, 1.0];

                }

                if (!palette) return null;

                lum = Math.max(0, Math.min(1, lum));

                let closestIdx = 0; let minDiff = Infinity;

                for (let i = 0; i < stops.length; i++) {

                    let diff = Math.abs(lum - stops[i]);

                    if (diff < minDiff) { minDiff = diff; closestIdx = i; }

                }

                return `rgb(${palette[closestIdx][0]},${palette[closestIdx][1]},${palette[closestIdx][2]})`;

            }


            function getInterpolatedGridSize(time) {

                if (keyframes.length === 0) return parseFloat(gridSize.value);

                const sorted = [...keyframes].sort((a, b) => a.time - b.time);

                if (time <= sorted[0].time) return sorted[0].value;

                if (time >= sorted[sorted.length - 1].time) return sorted[sorted.length - 1].value;

                for (let i = 0; i < sorted.length - 1; i++) {

                    const start = sorted[i]; const end = sorted[i+1];

                    if (time >= start.time && time <= end.time) {

                        const t = (time - start.time) / (end.time - start.time);

                        return start.value + (end.value - start.value) * t;

                    }

                }

                return parseFloat(gridSize.value);

            }


            function generateHalftone(targetCanvas, currentScale, source, gridValue) {

                if(!targetCanvas) return;

                const ctx = targetCanvas.getContext('2d', { alpha: false });

                const procW = targetCanvas.width;

                const procH = targetCanvas.height;


                let offscreen, oCtx;

                if (typeof OffscreenCanvas !== 'undefined') {

                    offscreen = new OffscreenCanvas(procW, procH); oCtx = offscreen.getContext('2d', { alpha: false });

                } else {

                    offscreen = document.createElement('canvas'); offscreen.width = procW; offscreen.height = procH; oCtx = offscreen.getContext('2d', { alpha: false });

                }

                

                oCtx.fillStyle = '#ffffff'; oCtx.fillRect(0, 0, procW, procH);

                

                const sW = source.videoWidth || source.width;

                const sH = source.videoHeight || source.height;

                const srcRatio = sW / sH; const dstRatio = procW / procH;

                let drawW = procW, drawH = procH, drawX = 0, drawY = 0;

                if (srcRatio > dstRatio) { drawH = procW / srcRatio; drawY = (procH - drawH) / 2; } 

                else { drawW = procH * srcRatio; drawX = (procW - drawW) / 2; }

                

                const srcB = parseInt(sourceBrightness.value, 10);

                const srcC = parseInt(sourceContrast.value, 10);

                oCtx.filter = `brightness(${100 + srcB}%) contrast(${100 + srcC}%)`;

                oCtx.drawImage(source, drawX, drawY, drawW, drawH);

                oCtx.filter = 'none';


                const imgData = oCtx.getImageData(0, 0, procW, procH);

                const data = imgData.data;


                const bAdj = parseInt(brightness.value, 10);

                const cAdj = parseInt(contrast.value, 10);

                const cFactor = (259 * (cAdj + 255)) / (255 * (259 - cAdj));


                const baseGrid = Math.max(1, gridValue * currentScale);

                const maxRadiusScale = parseFloat(dotScale.value);

                const isMappingInverted = invertMapping.checked;

                const useSmooth = smoothStep.checked;

                const rMode = currentRenderMode;

                const aCharsRaw = asciiCharsInput.value || " ";

                const aLogic = asciiLogicSelect ? asciiLogicSelect.value : 'density';

                const dStyle = dotStyleSelect.value;

                const isFixedSize = fixedDotSize.checked;

                const cutoff = parseFloat(dotCutoff.value);

                const aRatio = asciiRatio ? parseFloat(asciiRatio.value) : 0.6;

                const aIntensity = asciiIntensity ? parseInt(asciiIntensity.value) : 1;

                

                const removeSpaces = asciiRemoveSpaces ? asciiRemoveSpaces.checked : true;

                let safeChars = aCharsRaw;

                if (rMode === 'ascii' && aLogic === 'mask' && removeSpaces) {

                    safeChars = aCharsRaw.replace(/\s+/g, '');

                    if (safeChars.length === 0) safeChars = "MATRIX";

                }


                const stepX = (rMode === 'ascii') ? baseGrid * aRatio : baseGrid;

                const stepY = baseGrid;


                ctx.fillStyle = bgColorPicker.value;

                ctx.fillRect(0, 0, procW, procH);

                

                const numCols = Math.floor(procW / stepX);

                const numRows = Math.floor(procH / stepY);

                const offsetX = (procW - (numCols * stepX)) / 2;

                const offsetY = (procH - (numRows * stepY)) / 2;


                let textIdx = 0; 


                for (let row = 0; row < numRows; row++) {

                    for (let col = 0; col < numCols; col++) {

                        

                        let maskChar = '';

                        if (rMode === 'ascii' && aLogic === 'mask') {

                            maskChar = safeChars[textIdx % safeChars.length];

                            textIdx++;

                        }


                        let totalLum = 0, totalR = 0, totalG = 0, totalB = 0, count = 0;

                        const startX = Math.floor(offsetX + col * stepX);

                        const startY = Math.floor(offsetY + row * stepY);

                        const endX = Math.min(Math.floor(offsetX + (col + 1) * stepX), procW);

                        const endY = Math.min(Math.floor(offsetY + (row + 1) * stepY), procH);


                        for (let py = startY; py < endY; py++) {

                            for (let px = startX; px < endX; px++) {

                                const idx = (py * procW + px) * 4;

                                const r = data[idx], g = data[idx+1], b = data[idx+2];

                                let lum = 0.299 * r + 0.587 * g + 0.114 * b;

                                lum = cFactor * (lum - 128) + 128 + bAdj;

                                totalLum += Math.max(0, Math.min(255, lum));

                                if (dStyle === 'original') { totalR += r; totalG += g; totalB += b; }

                                count++;

                            }

                        }


                        if (count > 0) {

                            const avgLum = totalLum / count;

                            let normalizedLum = avgLum / 255;

                            let renderRatio = isMappingInverted ? normalizedLum : (1 - normalizedLum);


                            if (useSmooth) {

                                renderRatio = Math.max(0, Math.min(1, renderRatio));

                                renderRatio = renderRatio * renderRatio * (3 - 2 * renderRatio);

                            }


                            const centerX = offsetX + col * stepX + stepX / 2;

                            const centerY = offsetY + row * stepY + stepY / 2;


                            if (rMode === 'dot') {

                                const maxRadius = (stepY / 2) * maxRadiusScale;

                                let radius = 0;

                                if (isFixedSize) { if (renderRatio > cutoff) radius = maxRadius; } 

                                else { if (renderRatio > cutoff) radius = maxRadius * renderRatio; }


                                if (radius > 0.3) {

                                    if (dStyle === 'original') { ctx.fillStyle = `rgb(${Math.round(totalR/count)}, ${Math.round(totalG/count)}, ${Math.round(totalB/count)})`; } 

                                    else if (dStyle.startsWith('volumetric')) { ctx.fillStyle = getVolumetricColor(normalizedLum, dStyle); } 

                                    else { ctx.fillStyle = dotColorPicker.value; }

                                    ctx.beginPath(); ctx.arc(centerX, centerY, radius, 0, Math.PI * 2); ctx.fill();

                                }

                            } else {

                                if (dStyle === 'original') { ctx.fillStyle = `rgb(${Math.round(totalR/count)}, ${Math.round(totalG/count)}, ${Math.round(totalB/count)})`; } 

                                else if (dStyle.startsWith('volumetric')) { ctx.fillStyle = getVolumetricColor(normalizedLum, dStyle); } 

                                else { ctx.fillStyle = dotColorPicker.value; }


                                const fontSize = stepY * maxRadiusScale * 1.2;

                                ctx.font = `900 ${fontSize}px "JetBrains Mono", "Courier New", Courier, monospace`;

                                ctx.textAlign = 'center'; ctx.textBaseline = 'middle';


                                if (aLogic === 'mask') {

                                    if (renderRatio > cutoff && maskChar !== ' ') {

                                        ctx.globalAlpha = Math.max(0, Math.min(1, renderRatio));

                                        for(let i = 0; i < aIntensity; i++) {

                                            ctx.fillText(maskChar, centerX, centerY);

            if (typeof __firebase_config !== 'undefined') { cloudStatus.textContent = t('status_conn'); cloudStatus.className = "text-[9px] text-[var(--accent)] font-bold px-3 py-1 neu-inset rounded-full"; try { const firebaseConfig = JSON.parse(__firebase_config); currentAppId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id'; const app = initializeApp(firebaseConfig); const auth = getAuth(app); db = getFirestore(app); const initAuth = async () => { if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) await signInWithCustomToken(auth, __initial_auth_token); else await signInAnonymously(auth); }; initAuth().then(() => { onAuthStateChanged(auth, (user) => { currentUser = user; if (user) { onSnapshot(collection(db, 'artifacts', currentAppId, 'users', user.uid, 'presets'), (snapshot) => { userPresets = {}; snapshot.forEach(doc => userPresets[doc.id] = doc.data()); updatePresetDropdown(); cloudStatus.textContent = t('status_synced'); cloudStatus.className = "text-[9px] text-[var(--accent)] font-bold px-3 py-1 neu-inset rounded-full"; }, () => { cloudStatus.textContent = t('status_sync_err'); cloudStatus.className = "text-[9px] text-red-500 font-bold px-3 py-1 neu-inset rounded-full"; }); } }); }).catch(e => loadLocalPresets()); } catch (e) { loadLocalPresets(); } } else { loadLocalPresets(); }

            function loadLocalPresets() { cloudStatus.textContent = t('status_local'); cloudStatus.className = "text-[9px] text-[var(--text-muted)] font-bold px-3 py-1 neu-inset rounded-full"; try { userPresets = JSON.parse(localStorage.getItem('halftone_user_presets') || '{}'); } catch (e) { userPresets = {}; } updatePresetDropdown(); }


            presetSelect.addEventListener('change', (e) => {

                const name = e.target.value;

                if (name && userPresets[name]) {

                    const p = userPresets[name];

                    gridSize.value = p.gridSize; gridSizeInput.value = parseFloat(p.gridSize).toFixed(1); dotScale.value = p.dotScale; sourceBrightness.value = p.sourceBrightness; sourceContrast.value = p.sourceContrast; brightness.value = p.brightness; contrast.value = p.contrast; fpsInput.value = p.fps || defaults.fps; bgColorPicker.value = p.bgColor || defaults.bgColor; dotColorPicker.value = p.dotColor || defaults.dotColor; invertMapping.checked = p.invertMapping; smoothStep.checked = p.smoothStep !== undefined ? p.smoothStep : defaults.smoothStep; fixedDotSize.checked = p.fixedDotSize !== undefined ? p.fixedDotSize : defaults.fixedDotSize; dotCutoff.value = p.dotCutoff !== undefined ? p.dotCutoff : defaults.dotCutoff;

                    if (p.renderMode) setRenderMode(p.renderMode);

                    if (p.dotStyle) { dotStyleSelect.value = p.dotStyle; if (p.dotStyle === 'volumetric_custom') { customColorDark.value = p.customColorDark || "#000000"; customColorMid.value = p.customColorMid || "#99bbff"; customColorLight.value = p.customColorLight || "#ffffff"; } updateDotStyleUI(); }

                    if (p.asciiLogic) { asciiLogicSelect.value = p.asciiLogic; }

                    if (p.asciiRatio && asciiRatio) { asciiRatio.value = p.asciiRatio; }

                    if (p.asciiIntensity && asciiIntensity) { asciiIntensity.value = p.asciiIntensity; }

                    if (p.asciiRemoveSpaces !== undefined && asciiRemoveSpaces) { asciiRemoveSpaces.checked = p.asciiRemoveSpaces; }

                    if (p.asciiChars) { asciiCharsInput.value = p.asciiChars; const match = Array.from(asciiPresetSelect.options).find(opt => opt.value === p.asciiChars); asciiPresetSelect.value = match ? match.value : 'custom'; }

                    deletePresetBtn.classList.remove('hidden'); isManuallyOverridingGrid = false; updateUIValues(); if (!isVideo || (videoElement && videoElement.paused)) processFrame();

                } else { deletePresetBtn.classList.add('hidden'); }

            });


            savePresetBtn.addEventListener('click', async () => {

                const name = await showModal({ title: t('msg_save_preset'), message: t('msg_enter_name'), showInput: true }); if (!name) return;

                const preset = { gridSize: parseFloat(gridSize.value), dotScale: parseFloat(dotScale.value), sourceBrightness: parseInt(sourceBrightness.value), sourceContrast: parseInt(sourceContrast.value), brightness: parseInt(brightness.value), contrast: parseInt(contrast.value), fps: parseInt(fpsInput.value), bgColor: bgColorPicker.value, dotColor: dotColorPicker.value, invertMapping: invertMapping.checked, smoothStep: smoothStep.checked, fixedDotSize: fixedDotSize.checked, dotCutoff: parseFloat(dotCutoff.value), renderMode: currentRenderMode, asciiLogic: asciiLogicSelect.value, asciiRatio: asciiRatio ? parseFloat(asciiRatio.value) : 0.6, asciiIntensity: asciiIntensity ? parseInt(asciiIntensity.value) : 1, asciiRemoveSpaces: asciiRemoveSpaces ? asciiRemoveSpaces.checked : true, dotStyle: dotStyleSelect.value, customColorDark: customColorDark.value, customColorMid: customColorMid.value, customColorLight: customColorLight.value, asciiChars: asciiCharsInput.value };

                userPresets[name] = preset; presetSelect.value = name; deletePresetBtn.classList.remove('hidden');

                if (db && currentUser) { try { await setDoc(doc(db, 'artifacts', currentAppId, 'users', currentUser.uid, 'presets', name), preset); } catch (e) { showModal({ title: t('msg_err'), message: t('msg_err_cloud'), hideCancel: true }); } } else { localStorage.setItem('halftone_user_presets', JSON.stringify(userPresets)); updatePresetDropdown(); presetSelect.value = name; }

            });


            deletePresetBtn.addEventListener('click', async () => {

                const name = presetSelect.value; if (!name) return;

                if (await showModal({ title: t('msg_del_preset'), message: `${t('msg_confirm_del')} "${name}" ?`, showInput: false })) {

                    delete userPresets[name]; presetSelect.value = ""; deletePresetBtn.classList.add('hidden');

                    if (db && currentUser) { try { await deleteDoc(doc(db, 'artifacts', currentAppId, 'users', currentUser.uid, 'presets', name)); } catch (e) {} } else { localStorage.setItem('halftone_user_presets', JSON.stringify(userPresets)); updatePresetDropdown(); }

                }

            });


            // Initial calls

            bgColorPicker.value = defaults.bgColor; dotStyleSelect.value = defaults.dotStyle; contrast.value = defaults.contrast; fixedDotSize.checked = defaults.fixedDotSize; dotCutoff.value = defaults.dotCutoff;

            applyI18n(); // Load language first

            updateDotStyleUI(); updateUIValues();

        });
